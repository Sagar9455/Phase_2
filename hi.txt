#!/bin/bash

set -e

echo "======================================="
echo "   One-Click Build Script (RPI)"
echo "======================================="

PROJECT_ROOT="$(pwd)"
APP_PATH="$PROJECT_ROOT/input/application/main.py"

# Step 1: Clean old builds

echo "[1/7] Cleaning old builds..."
rm -rf build dist *.spec **pycache**

# Step 2: Update system

echo "[2/7] Updating system..."
sudo apt update -y

# Step 3: Install system dependencies (Python + Pillow libs + build tools)

echo "[3/7] Installing system dependencies..."
sudo apt install -y 
python3 
python3-pip 
python3-dev 
build-essential 
libjpeg-dev 
zlib1g-dev 
libfreetype6-dev 
liblcms2-dev 
libopenjp2-7-dev 
libtiff5-dev 
libwebp-dev 
tcl8.6-dev 
tk8.6-dev

# Step 4: Upgrade pip tools

echo "[4/7] Upgrading pip tools..."
pip3 install --upgrade pip setuptools wheel

# Step 5: Install Python dependencies

echo "[5/7] Installing Python dependencies..."
if [ -f "requirements.txt" ]; then
pip3 install --no-cache-dir -r requirements.txt
else
echo "No requirements.txt found, skipping..."
fi

# Step 6: Install PyInstaller

echo "[6/7] Installing PyInstaller..."
pip3 install pyinstaller

# Step 7: Build executable

echo "[7/7] Building executable..."

pyinstaller --onefile "$APP_PATH" 
--paths="$PROJECT_ROOT/input" 
--add-data "$PROJECT_ROOT/input/application/config.json:." 
--add-data "$PROJECT_ROOT/input/supportfiles:./supportfiles" 
--add-data "$PROJECT_ROOT/input/drivers:./drivers" 
--hidden-import=can 
--hidden-import=can.interface 
--hidden-import=can.interfaces.socketcan 
--collect-all can

echo "======================================="
echo " Build completed successfully!"
echo " Executable: dist/main"
echo " Run using: ./dist/main"
echo "======================================="
