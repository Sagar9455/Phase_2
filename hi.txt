#!/bin/bash
# ============================================================
#  UDS Diagnostics - Raspberry Pi Build Script (FAST VERSION)
# ============================================================

set -e

PROJECT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
APP_NAME="uds_diagnostics"
OUTPUT_DIR="$PROJECT_DIR/dist"
VENV_DIR="$PROJECT_DIR/venv"

echo ""
echo "=============================================="
echo "  UDS Diagnostics - Fast Build Tool"
echo "=============================================="
echo ""

# --- Step 1: Check Python ---
echo "[1/7] Checking Python..."
python3 --version || { echo "ERROR: Install python3"; exit 1; }

# --- Step 2: Ensure pip + venv ---
echo "[2/7] Checking pip & venv..."
python3 -m pip --version 2>/dev/null || {
    echo "Installing pip & venv..."
    sudo apt-get update
    sudo apt-get install -y python3-pip python3-venv
}

# --- Step 3: Create venv (only once) ---
echo "[3/7] Setting up virtual environment..."
if [ ! -d "$VENV_DIR" ]; then
    python3 -m venv "$VENV_DIR"
    echo "  ✔ venv created"
else
    echo "  ✔ venv exists (skipping)"
fi

# --- Step 4: Activate venv ---
echo "[4/7] Activating venv..."
source "$VENV_DIR/bin/activate"

# --- Step 5: Install dependencies (ONLY FIRST TIME) ---
echo "[5/7] Checking dependencies..."
if [ ! -f "$VENV_DIR/.deps_installed" ]; then
    echo "  Installing dependencies (first run)..."
    
    pip install --upgrade pip

    pip install \
        pyinstaller \
        RPi.GPIO \
        adafruit-circuitpython-ssd1306 \
        adafruit-blinka \
        Pillow \
        python-can \
        can-isotp \
        udsoncan

    touch "$VENV_DIR/.deps_installed"
else
    echo "  ✔ Dependencies already installed (fast mode)"
fi

# --- Step 6: Prepare project ---
echo "[6/7] Preparing project..."
cd "$PROJECT_DIR"

# Ensure package structure
touch drivers/__init__.py

# Optional clean mode
if [ "$1" == "clean" ]; then
    echo "  Cleaning old build..."
    rm -rf build dist *.spec
fi

# --- Step 7: Build executable ---
echo "[7/7] Building executable..."
echo "      (Fast rebuild enabled)"
echo ""

pyinstaller \
    --noconfirm \
    --onefile \
    --name "$APP_NAME" \
    --add-data "config.json:." \
    --add-data "drivers:drivers" \
    --hidden-import RPi \
    --hidden-import RPi.GPIO \
    --hidden-import board \
    --hidden-import busio \
    --hidden-import adafruit_ssd1306 \
    --hidden-import PIL \
    --hidden-import PIL.Image \
    --hidden-import PIL.ImageDraw \
    --hidden-import PIL.ImageFont \
    --hidden-import can \
    --hidden-import can.interfaces.socketcan \
    --hidden-import can.io.asc \
    --hidden-import isotp \
    --hidden-import udsoncan \
    --hidden-import udsoncan.client \
    --hidden-import udsoncan.connections \
    --hidden-import udsoncan.configs \
    --hidden-import udsoncan.services \
    --hidden-import drivers \
    --hidden-import drivers.config_loader \
    --hidden-import drivers.oled_display \
    --hidden-import drivers.button_input \
    --hidden-import drivers.uds_client \
    --hidden-import drivers.transfer_file \
    --hidden-import drivers.Parse_handler \
    --hidden-import drivers.can_logger \
    --hidden-import drivers.report_generator \
    main.py

# --- Done ---
echo ""
echo "=============================================="
echo "  Build complete!"
echo "=============================================="
echo ""

if [ -f "$OUTPUT_DIR/$APP_NAME" ]; then
    echo "  ✅ Executable created:"
    echo "     $OUTPUT_DIR/$APP_NAME"
    echo ""
    echo "  Size: $(du -sh "$OUTPUT_DIR/$APP_NAME" | cut -f1)"
    echo ""
    echo "  Run with:"
    echo "     sudo $OUTPUT_DIR/$APP_NAME"
else
    echo "  ❌ Build failed"
    exit 1
fi

# Deactivate venv
deactivate
